def get_db_connection():
    try:
        scope = ["https://spreadsheets.google.com/feeds", "https://www.googleapis.com/auth/drive"]
        creds = ServiceAccountCredentials.from_json_keyfile_dict(st.secrets["gcp_service_account"], scope)
        client = gspread.authorize(creds)
        sheet = client.open("Idiomaconnect_DB").sheet1
        return sheet
    except Exception as e:
        return None

def generate_lesson(profile_name, topic, custom_audio_text=None):
    if not groq_client:
        return "⚠️ API de Groq no configurada. Por favor, añade tu API Key en secrets."
    
    profile = PROFILES[profile_name]
    
    system_prompt = f"""
    Eres un tutor de inglés experto, cariñoso y motivador, diseñado exclusivamente para {profile_name}, una niña de 13 años (nacida el 24/Sept/2012).
    A ella le apasiona: {profile['hobbies']}.
    Tu tono debe ser: {profile['tone']}.
    
    {FAMILY_CONTEXT}
    
    INSTRUCCIONES:
    - Crea una lección corta, interactiva y gamificada.
    - Integra al menos 1 o 2 miembros de su familia o mascotas en los ejemplos en inglés.
    - Incluye un pequeño desafío o pregunta al final para que gane XP.
    - Mantén el formato limpio, usando viñetas y emojis. Escribe en Spanglish (explicaciones en español, vocabulario en inglés).
    """
    
    user_prompt = f"El tema de hoy es: {topic}."
    if custom_audio_text:
        user_prompt += f" La alumna ha dicho por voz: '{custom_audio_text}'. Adapta la lección a esto."

    response = groq_client.chat.completions.create(
        messages=[
            {"role": "system", "content": system_prompt},
            {"role": "user", "content": user_prompt}
        ],
        model="llama3-8b-8192",
        temperature=0.7,
        max_tokens=1000,
    )
    return response.choices[0].message.content

def send_weekly_report():
    # Lógica de envío de correo (solo los viernes)
    today = datetime.datetime.now()
    if today.weekday() == 4: # 4 = Viernes
        try:
            msg = MIMEMultipart()
            msg['From'] = st.secrets["email_sender"]
            msg['To'] = "jccubillos@gmail.com"
            msg['Subject'] = "Reporte Semanal Idiomaconnect 🚀"
            
            # Aquí idealmente leeríamos de GSheets, esto es un mockup de los datos
            body = f"¡Hola Juan Carlos!\n\nEste es el progreso de las trillizas esta semana:\n- Antonia: 150 XP ganados.\n- Belén: 120 XP ganados.\n- Sofía: 180 XP ganados.\n\n¡Van excelente!"
            msg.attach(MIMEText(body, 'plain'))
            
            server = smtplib.SMTP('smtp.gmail.com', 587)
            server.starttls()
            server.login(st.secrets["email_sender"], st.secrets["email_password"])
            text = msg.as_string()
            server.sendmail(st.secrets["email_sender"], "jccubillos@gmail.com", text)
            server.quit()
        except Exception as e:
            pass # Falla silenciosa si no hay credenciales
